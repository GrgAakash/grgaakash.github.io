<script>
document.addEventListener('DOMContentLoaded', function() {
    // Enhanced modal with loading state
    const imageModal = document.getElementById('imageModal');
    if (imageModal) {
        imageModal.addEventListener('show.bs.modal', function(event) {
            const button = event.relatedTarget;
            const imageSrc = button.getAttribute('data-image');
            const imageTitle = button.getAttribute('data-title');

            const modalImage = imageModal.querySelector('#modalImage');
            const modalTitle = imageModal.querySelector('#imageModalLabel');
            const modalBody = imageModal.querySelector('.modal-body');

            // Add loading state
            if (modalBody) {
                modalBody.classList.add('loading');
            }

            // Remove loaded class and reset transform
            if (modalImage) {
                modalImage.classList.remove('loaded');
                modalImage.src = '';

                // Create new image to preload
                const img = new Image();
                img.onload = function() {
                    modalImage.src = imageSrc;
                    // Remove loading state and add loaded class after a short delay
                    setTimeout(function() {
                        if (modalBody) {
                            modalBody.classList.remove('loading');
                        }
                        modalImage.classList.add('loaded');
                    }, 100);
                };
                img.onerror = function() {
                    // Remove loading state even on error
                    if (modalBody) {
                        modalBody.classList.remove('loading');
                    }
                    modalImage.src = imageSrc;
                    modalImage.classList.add('loaded');
                };
                img.src = imageSrc;
            }

            if (modalTitle) {
                modalTitle.textContent = imageTitle;
            }
        });

        // Reset modal state on hide
        imageModal.addEventListener('hidden.bs.modal', function() {
            const modalImage = imageModal.querySelector('#modalImage');
            const modalBody = imageModal.querySelector('.modal-body');

            if (modalImage) {
                modalImage.classList.remove('loaded');
            }
            if (modalBody) {
                modalBody.classList.remove('loading');
            }
        });
    }

    // Video play state management
    const videos = document.querySelectorAll('.gallery-video');
    videos.forEach(function(video) {
        const container = video.closest('.video-container');

        video.addEventListener('play', function() {
            if (container) {
                container.classList.add('playing');
            }
        });

        video.addEventListener('pause', function() {
            if (container) {
                container.classList.remove('playing');
            }
        });

        video.addEventListener('ended', function() {
            if (container) {
                container.classList.remove('playing');
            }
        });
    });

    // Intersection Observer for lazy animations (optional enhancement)
    // This can be used if you want animations to trigger only when items are in viewport
    if ('IntersectionObserver' in window) {
        const galleryItems = document.querySelectorAll('#gallery .col-md-6');

        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -50px 0px'
        };

        const observer = new IntersectionObserver(function(entries) {
            entries.forEach(function(entry) {
                if (entry.isIntersecting) {
                    entry.target.style.animationPlayState = 'running';
                    observer.unobserve(entry.target);
                }
            });
        }, observerOptions);

        // Note: Intersection Observer is ready but not actively pausing animations
        // since the initial page load animation looks good. Uncomment below to enable lazy loading.
        // galleryItems.forEach(function(item) {
        //     item.style.animationPlayState = 'paused';
        //     observer.observe(item);
        // });
    }
});
</script>
